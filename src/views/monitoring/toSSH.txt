  import { socket } from "../../constants.js";
import { Terminal } from "xterm";
import { WebLinksAddon } from "xterm-addon-web-links";
import * as fit from "xterm/lib/addons/fit/fit";
  sockets: {
    sshConnectStage(objFromServer) {
      console.log("SSH connect");
      this.term.write("\r\n*** Connected to backend***\r\n");

      // Browser -> Backend
      this.term.on("data", data => {
        this.$socket.emit("sshCommand", data);
      });
    },
    // Backend -> Browser
    sshOutput(data) {
      console.log(data);
      this.term.write(data);
    },
    sshDisconnect() {
      this.term.write("\r\n*** Disconnected from backend***\r\n");
    }
  },

  methods: {
    connectSSH(ip) {
      const query = {
        ip: ip,
        password: this.sshPassword,
        cmd: ""
      };
      socket.emit("sshConnect", query);
    }
  },
    data() {
    return {
      sshPassword: "",
      sshOutput: ""
    };
  },
   async mounted() {
    
    const terminalContainer = document.getElementById("terminal-container");
    Terminal.applyAddon(fit);
    this.term = new Terminal({ cursorBlink: true });
    this.term.loadAddon(new WebLinksAddon());
    this.term.open(terminalContainer);
    this.term.fit();
